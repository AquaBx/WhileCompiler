cmake_minimum_required(VERSION 3.29)
project(TextEditor)
set (CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
set (CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
set(CMAKE_CXX_STANDARD 26)

# Tests

enable_testing()

add_custom_command ( 
        OUTPUT bools.cpp
        COMMAND java -jar ${CMAKE_CURRENT_SOURCE_DIR}/src/while_compiler/target/while_compiler-all.jar -i lang/bools.while -o out/bools.cpp
        DEPENDS lang/bools.while
        WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/test
)

add_executable(bools bools.cpp)
add_test(NAME TestBools COMMAND bools)
